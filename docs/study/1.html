Ex = sum(x * pro_x) # 기댓값
Exx = sum(x**2 * pro_x) # E[X^2]

# Var(X) = E[X^2] - (E[X])^2
Exx - Ex**2

# or E[(X - E[X])**2]
sum((x - Ex)**2 * pro_x)
